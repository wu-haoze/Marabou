macro(constraint_add_unit_test name)
    set(USE_MOCK_COMMON FALSE)
    set(USE_MOCK_ENGINE FALSE)
    marabou_add_test(${CONSTRAINT_TESTS_DIR}/Test_${name} constraint USE_MOCK_COMMON
        USE_MOCK_ENGINE "unit")
endmacro()
